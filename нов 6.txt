create or replace PROCEDURE SEARCH_Between_DATES
(V_datesale product_sale.datesale %TYPE,
v_datesale1 product_sale.datesale%type)
IS
BEGIN
DECLARE
CURSOR Purvi IS
SELECT pr.sale_id,pr.id_client,pr.id_employee,pr.datesale 
FROM product_sale pr
join shopping_card sh on sh.sale_id = pr.sale_id
join products p on p.id_product=sh.id_product
WHERE pr.datesale Between  v_datesale AND  v_datesale1 ;
BEGIN
FOR CUST_RECORD IN Purvi
LOOP
DBMS_OUTPUT.PUT_LINE(CUST_RECORD.sale_id||' '||CUST_RECORD.id_client||' '||CUST_RECORD.id_employee||' '||CUST_RECORD.datesale);
DBMS_OUTPUT.new_LINE();
END LOOP;
end;
end;


CREATE OR REPLACE PROCEDURE SEARCH_SALE_BY_CLIENT
(V_NAMEE clientt.NAMEE %TYPE)
IS
BEGIN
DECLARE
CURSOR Purvi IS
SELECT cli.namee,pr.datesale,e.id_client
FROM clientt cli
join product_sale pr on pr.id_client = cli.id_client
join shopping_card sh on sh.sale_id = pr.sale_id
WHERE e.namee = V_NAMEE;
BEGIN
FOR CUST_RECORD IN Purvi
LOOP
DBMS_OUTPUT.PUT_LINE(CUST_RECORD.namee||' '||CUST_RECORD.ID_CLIENT);
END LOOP;
END;
END;
